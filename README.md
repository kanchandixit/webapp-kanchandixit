2. TERRAFORM SAMPLE FOLDER (READY TO UPLOAD)

Create folder:

infra/

infra/main.tf
terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
      version = "~> 3.0"
    }
  }

  required_version = ">= 1.0"
}

provider "azurerm" {
  features {}
}

resource "azurerm_resource_group" "rg" {
  name     = "rg-kanchadixit"
  location = "EastUS"
}

infra/variables.tf
variable "location" {
  default = "EastUS"
}

infra/outputs.tf
output "resource_group_name" {
  value = azurerm_resource_group.rg.name
}
